//TRAWL LOG SCRIPT - quering a view created FROM MARFIS DATA

SET LINESIZE 300
set pagesize 1000

select	VESSEL_NAME VESSEL,VR_NUMBER BCODE, LICENCE_ID,TRIP_NUMBER TRIP#, TOW TOW#, 
	DATE_FISHED FDATE, DATE_LANDED LDATE, LATITUDE BEG_LAT, LONGITUDE B_LONG,END_LATITUDE END_LAT,
	END_LONGITUDE E_LONG, DET_SHRIMP_FISHING_AREA SFA, AVERAGE_DEPTH DEPTH, TOW_DURATION_HOURS HRS,
	TOW_DURATION_MINUTES MINS,RND_WEIGHT_KGS TOW_WGT_KGS, COUNT_PER_LB xCOUNT
from	MARFISSCI.MOBILE_SHRIMP 
where	DET_SHRIMP_FISHING_AREA = 16
ORDER BY VESSEL_NAME, DATE_FISHED, TRIP_NUMBER, TOW;
